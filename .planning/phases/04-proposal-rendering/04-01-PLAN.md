**1a. Create src/lib/proposal/url-params.ts**

Implement two functions following the research-confirmed base64 pattern (handles all Unicode in customer names/addresses):

```typescript
import type { ProposalFormValues } from '@/lib/form/schema';

export function encodeProposalParams(values: ProposalFormValues): string {
  return btoa(encodeURIComponent(JSON.stringify(values)));
}

export function decodeProposalParams(encoded: string): ProposalFormValues {
  return JSON.parse(decodeURIComponent(atob(encoded))) as ProposalFormValues;
}
```

Structure of the rendered JSX:
```tsx
import './proposal.css';

export default async function ProposalPage({ searchParams }: { searchParams: Promise<{ d?: string }> }) {
  const params = await searchParams;
  if (!params.d) redirect('/enter');

  let formValues: ProposalFormValues;
  try {
    formValues = decodeProposalParams(params.d);
  } catch {
    redirect('/enter');
  }

  const config = await getAlbertaConfig();
  const inputs = buildProposalInputs(formValues);  // local helper, same logic as runCalculation
  const outputs = calculateProposal(inputs, config);
  const serialized = serializeOutputs(outputs);    // local helper

  const satelliteUrl = buildSatelliteImageUrl(formValues.customer.address);

  return (
    <div className="proposal-root proposal-snap-container">
      {/* Pages 1-11 rendered here — placeholder divs for now, filled by plans 02 and 03 */}
      <div className="proposal-page flex items-center justify-center">
        <p style={{ color: 'var(--nrg-green)', fontFamily: 'var(--font-display)', fontSize: '2rem' }}>
          PROPOSAL LOADING — {formValues.customer.name}
        </p>
      </div>
    </div>
  );
}
```

Include in page.tsx:
- `buildSatelliteImageUrl(address: string): string | null` — constructs Maps Static API URL using `process.env.GOOGLE_MAPS_API_KEY` (server only); returns null if key absent or address empty. URL: `https://maps.googleapis.com/maps/api/staticmap?center={encoded}&zoom=19&size=640x400&scale=2&maptype=satellite&key={key}`
- `buildProposalInputs(formValues: ProposalFormValues): ProposalInputs` — identical to what runCalculation does internally, converts all string fields to Decimal
- `serializeOutputs(outputs: ProposalOutputs): SerializedProposalOutputs` — maps all Decimal fields to .toString()
