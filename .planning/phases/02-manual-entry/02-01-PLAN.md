  <action>
    Create `src/app/actions/get-defaults.ts` as a Next.js 15 Server Action:

    ```typescript
    'use server';
    import { getAlbertaConfig } from '@/lib/config/alberta';

    export interface DefaultRates {
      allInRate: string;
      netMeteringBuyRate: string;
      netMeteringSellRate: string;
      annualEscalationRate: string;
    }

    export async function getDefaultRates(): Promise<DefaultRates> {
      const config = await getAlbertaConfig();
      return {
        allInRate: config.defaultNetMeteringBuyRate.toString(),
        netMeteringBuyRate: config.defaultNetMeteringBuyRate.toString(),
        netMeteringSellRate: config.defaultNetMeteringSellRate.toString(),
        annualEscalationRate: config.defaultEscalationRate.toString(),
      };
    }
    ```

    IMPORTANT: Use `.toString()` on Decimal values to get clean string output — DO NOT pass through `Number()` or any floating-point conversion (Supabase NUMERIC → Decimal → toString preserves precision per Phase 1 pattern).

    Note: allInRate and netMeteringBuyRate are both initialized from defaultNetMeteringBuyRate — they represent the same rate default. The user can differentiate them after pre-population if needed.
