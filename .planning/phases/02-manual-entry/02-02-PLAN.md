    MonthlyGrid is the most complex component. It receives react-hook-form's `control` and `setValue`/`watch` from the parent form, managing 12 Controller-based monthly inputs plus an annual total input.

    ```typescript
    import { Control, UseFormSetValue, UseFormWatch } from 'react-hook-form';
    import { ProposalFormValues } from '@/lib/form/schema';
    import { ALBERTA_SOLAR_CURVE } from '@/lib/form/defaults';

    export interface MonthlyGridProps {
      control: Control<ProposalFormValues>;
      setValue: UseFormSetValue<ProposalFormValues>;
      watch: UseFormWatch<ProposalFormValues>;
      // Field path prefix: 'system' | 'consumption'
      section: 'system' | 'consumption';
      // Field names within the section
      annualFieldName: 'annualProductionKwh' | 'annualConsumptionKwh';
      monthlyFieldName: 'monthlyProductionKwh' | 'monthlyConsumptionKwh';
      label: string;                     // e.g. "Production (kWh)"
      useAlbertaCurve?: boolean;         // true for production, false for consumption (even split)
      errors?: Record<string, { message?: string }[]>; // Monthly field errors from formState
    }
    ```

    Layout:
    - Annual total input at top: full width, using FormInput component with label "Annual Total (kWh)"
    - Below annual: 4-column grid of 3 rows (months Jan-Mar | Apr-Jun | Jul-Sep | Oct-Dec)
    - Each monthly cell: month label (Jan, Feb, ...) above a compact FormInput (no label prop needed — month name serves as label)
    - Monthly inputs are compact: `text-xs py-1.5` override classes
    - Grid classes: `grid grid-cols-4 gap-2 mt-3`

    CRITICAL — Use Controller for every monthly input (not register):
    ```typescript
    // Wrong — programmatic setValue won't update displayed value:
    // <input {...register(`system.monthlyProductionKwh.${i}`)} />

    // Correct — Controller renders a React-controlled input:
    <Controller
      control={control}
      name={`${section}.${monthlyFieldName}.${i}` as any}
      render={({ field }) => (
        <FormInput
          label={MONTH_NAMES[i]}
          value={field.value}
          onChange={field.onChange}
          onBlur={field.onBlur}
          error={errors?.[i]?.message}
        />
      )}
    />
    ```

    Two-way sync with loop prevention:
    ```typescript
    const syncSource = useRef<'annual' | 'monthly' | null>(null);
    const annualPath = `${section}.${annualFieldName}` as const;
    const monthlyBasePath = `${section}.${monthlyFieldName}` as const;

    // Annual → monthly distribution
    useEffect(() => {
      const sub = watch((values, { name }) => {
        if (name !== annualPath) return;
        if (syncSource.current === 'monthly') return;
        const raw = (values as any)?.[section]?.[annualFieldName] ?? '';
        const annual = Number(raw);
        if (isNaN(annual) || annual <= 0) return;
        syncSource.current = 'annual';
        const curve = useAlbertaCurve ? ALBERTA_SOLAR_CURVE : Array(12).fill(1 / 12);
        curve.forEach((fraction, i) => {
          setValue(
            `${monthlyBasePath}.${i}` as any,
            (annual * fraction).toFixed(0),
            { shouldDirty: true, shouldValidate: false }
          );
        });
        syncSource.current = null;
      });
      return () => sub.unsubscribe();
    }, [watch, setValue, annualPath, monthlyBasePath, section, annualFieldName, useAlbertaCurve]);

    // Monthly → annual sum
    useEffect(() => {
      const sub = watch((values, { name }) => {
        if (!name?.startsWith(monthlyBasePath)) return;
        if (syncSource.current === 'annual') return;
        const months = (values as any)?.[section]?.[monthlyFieldName] ?? [];
        const sum = (months as string[]).reduce((acc, v) => acc + (Number(v) || 0), 0);
        syncSource.current = 'monthly';
        setValue(annualPath as any, sum.toFixed(0), { shouldDirty: true, shouldValidate: false });
        syncSource.current = null;
      });
      return () => sub.unsubscribe();
    }, [watch, setValue, annualPath, monthlyBasePath, section, monthlyFieldName]);
    ```

    Month name constants:
    ```typescript
    const MONTH_NAMES = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'] as const;
    ```

    Consumption distribution note: When `useAlbertaCurve` is false, use even 1/12 split (uniform monthly distribution for consumption — no Alberta-specific curve exists for consumption loads).
