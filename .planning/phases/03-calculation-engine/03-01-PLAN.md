**utility-projection.ts:**
```typescript
import { d } from '@/lib/decimal';
import type Decimal from 'decimal.js';

/** CALC-01: Returns yearly utility costs, length = years, index 0 = Year 1 */
export function projectUtilityCosts(annualCost: Decimal, escalationRate: Decimal, years: number): Decimal[] {
  return Array.from({ length: years }, (_, i) => {
    const n = i + 1;
    return annualCost.times(d('1').plus(escalationRate).pow(n));
  });
}
```
